  3: /**
  4:  * The SQL-based NoSQL library for PHP -- it's NoSQL, even though it's not!
  5:  *
  6:  * Uses some simplistic middleware logic to create a persistent key/value store from a PDO connection.
  7:  *
  8:  * Keys can be a string with anything in it, and values can be of any type, as they are converted to and from
  9:  * json on the fly. To take advantage of some minor performance-enhancing logic, keys should be stated like this:
 10:  *      foo.bar.baz
 11:  *
 12:  * The store has some limitations. For one, saving an object as users.admins.Heavenquake will mean that in order
 13:  * to retrieve the admin user called Heavenquake, you will have to enter that exact key into ->get(). If you would
 14:  * rather fetch a list of admins, you need to instead have a key called users.admins and then fill that with a PHP
 15:  * array of admin users. Think of this as less of a database solution and more as a .json file with increased
 16:  * performance and reliability in high concurrency deployments.
 17:  *
 18:  * https://github.com/sikker/not-nosql
 19:  *
 20:  * include 'notnosql/notnosql.class.php';
 21:  * $notnosql = new NotNoSQL( new PDO('sqlite:database.sq3') );
 22:  * $notnosql->put('foo', 'bar');
 23:  * echo $notnosql->get('foo') . PHP_EOL;
 24:  *
 25:  * @author  Per Sikker Hansen <persikkerhansen@gmail.com>
 26:  * @license Creative Commons Attribution License v.3.0. http://creativecommons.org/licenses/by/3.0/
 27:  * @package Tool
 28:  * @date 26.04.2016
 29:  */

 35:     private $db;
 36:     private $collections = array();
 37:     private $jsonDecodePolicy;
 38: 
 39:     const ROOT_COLLECTION = 'notnosql_root_collection';
 40:     const JSON_DECODE_POLICY_ARRAY = 1;
 41:     const JSON_DECODE_POLICY_OBJECT = 2;
 42: 
 43:     /**
 44:      * Create the store
 45:      *
 46:      * @param   PDO the database object the class will write to.
 47:      */
 48:     public function __construct(\PDO $db) {
 49:         $this->db = $db;
 50:         $this->setJsonDecodePolicy(self::JSON_DECODE_POLICY_ARRAY);
 51:     }
 52: 
 53:     /**
 54:      * Get a value
 55:      *
 56:      * @param   string
 57:      * @return  mixed   Returns null if no such key.
 58:      */
 59:     public function get($key) {
 60:         // We'll take the first segment of the key and use it as the collection name (ie ~users~.admins.foo.bar etc.)
 61:         $segments = explode('.', $key);
 62:         if (count($segments) === 1) {
 63:             // If there isn't a first segment, we'll put it in the hidden root collection
 64:             return $this->statement('get', self::ROOT_COLLECTION, $key);
 65:         } else {
 66:             // If there is, we'll use the rest of the given key as the new key.
 67:             $collection = array_shift($segments);
 68:             return $this->statement('get', $collection, implode('.', $segments));
 69:         }
 70:     }
 71: 
 72:     /**
 73:      * Save a value
 74:      *
 75:      * @param   string
 76:      * @param   mixed   Anything goes. Will get json_encoded anyway.
 77:      * @return  bool
 78:      */
 79:     public function put($key, $value) {
 80:         // We'll take the first segment of the key and use it as the collection name (ie ~users~.admins.foo.bar.baz etc.)
 81:         $segments = explode('.', $key);
 82:         if (count($segments) === 1) {
 83:             // If there isn't a first segment, we'll put it in the hidden root collection
 84:             return $this->statement('put', self::ROOT_COLLECTION, $key, $value);
 85:         } else {
 86:             // If there is, we'll use the rest of the given key as the new key.
 87:             $collection = array_shift($segments);
 88:             return $this->statement('put', $collection, implode('.', $segments), $value);
 89:         }
 90:     }
 91: 
 92:     /**
 93:      * Change the policy for decoding JSON objects.
 94:      *
 95:      * By default get() calls will, for performance reasons, return JSON objects as associative arrays, but your code
 96:      * may require PHP object returns, in which case this method is offered.
 97:      *
 98:      * @param   int NotNoSQL::JSON_DECODE_POLICY_ARRAY or NotNoSQL::JSON_DECODE_POLICY_OBJECT
 99:      * @return  void
100:      */
101:     public function setJsonDecodePolicy($jsonDecodePolicy) {
102:         if ($jsonDecodePolicy !== self::JSON_DECODE_POLICY_ARRAY && $jsonDecodePolicy !== self::JSON_DECODE_POLICY_OBJECT) {
103:             throw new NotNoSQLException('Invalid jsonDecodePolicy value');
104:         }
105:         $this->jsonDecodePolicy = $jsonDecodePolicy;
106:     }
107: 
108:     /**
109:      * Check what policy is currently used for converting JSON objects.
110:      *
111:      * @return  int NotNoSQL::JSON_DECODE_POLICY_ARRAY or NotNoSQL::JSON_DECODE_POLICY_OBJECT
112:      */
113:     public function getJsonDecodePolicy() {
114:         return $this->jsonDecodePolicy;
115:     }
116: 
117:     // -------------------------------------------------------------------------------------------------------------------
118: 
119:     /**
120:      * Helper method for invoking the correct prepared statement, as well as creating new collections
121:      *
122:      * @param   string  'get' or 'put' -- it automatically checks if it's a insert or update statement
123:      * @param   string
124:      * @param   string
125:      * @param   string  Optional. Only applicable for put statements
126:      * @return  mixed   The decoded value on get statements and true/false on put statements
127:      */
128:     private function statement($statement, $table, $key, $value = null) {
129:         // Make sure the collection exists
130:         $this->createStatement($table);
131: 
132:         switch ($statement) {
133:             case 'get':
134:                 // In a get scenario, no such key is a perfectly valid response, and will be turned into null.
135:                 try {
136:                     // The result is a valid PHP variable, decoded from json.
137:                     $result = $this->selectStatement($table, $key);
138:                 } catch (NotNoSQLException $e) {
139:                     $result = null;
140:                 }
141:                 return $result;
142:                 break;
143: 
144:             case 'put':
145:                 // While the put method doesn't care whether the key exists originally or not, sadly so does most databases.
146:                 try {
147:                     $this->foo = 'set';
148:                     $this->selectStatement($table, $key);
149:                     return $this->updateStatement($table, $key, $value);
150:                 } catch (NotNoSQLException $e) {
151:                     return $this->insertStatement($table, $key, $value);
152:                 }
153:                 break;
154:         }
155:     }
156: 
157:     /**
158:      * Creates and remembers a table
159:      *
160:      * @param   string
161:      * @return  void
162:      */
163:     private function createStatement($table) {
164:         if (isset($this->collections[$table])) {
165:             // No need to spam the CREATE statements if we've already tried
166:             return;
167:         }
168:         $result = $this->db->query(
169:             'CREATE TABLE IF NOT EXISTS `' . $table . '` (`key` TEXT, `value` TEXT)'
170:         );
171:         // Either it didn't exist and we created it, or it already existed. Either way, remember that we're done here.
172:         $this->collections[$table] = true;
173:     }
174: 
175:     /**
176:      * Selects data from a collection
177:      *
178:      * @param   string
179:      * @param   string
180:      * @return  mixed   Decoded json value
181:      */
182:     private function selectStatement($table, $key)
183:     {
184:         $buildQuery = 'SELECT `value` FROM `' . $table . '` WHERE `key` = "' . $this->escape($key) . '"';
185: 
186:         $result = $this->db->query($buildQuery)->fetchColumn();
187: 
188:         if ($result === false) {
189:             // No such key can mean different things, so instead of returning null we let the recipient deal with it.
190:             throw new NotNoSQLException('NO SUCH KEY');
191:         } else {
192:             // The value is decoded using the policy decided on with setJsonDecodePolicy. Default is associative array.
193:             return json_decode($this->unEscape($result), (
194:             $this->getJsonDecodePolicy() === self::JSON_DECODE_POLICY_ARRAY ?
195:                 true :
196:                 false
197:             ));
198:         }
199:     }
200: 
201:     /**
202:      * Inserts a new row in a collection
203:      *
204:      * @param   string
205:      * @param   string
206:      * @param   mixed   Anything goes. Will get transformed into JSON on the fly.
207:      * @return  void
208:      */
209:     private function insertStatement($table, $key, $value) {
210:         $this->db->query(
211:             'INSERT INTO `' . $table . '` (`key`, `value`) VALUES("' . $this->escape($key) . '", "' . $this->escape(json_encode($value)) . '")'
212:         );
213:     }
214: 
215:     /**
216:      * Updates a row in a collection
217:      *
218:      * @param   string
219:      * @param   string
220:      * @param   mixed   Anything goes. Will get transformed into JSON on the fly.
221:      * @return  void
222:      */
223:     private function updateStatement($table, $key, $value) {
224:         $this->db->query(
225:             'UPDATE `' . $table . '` SET `value` = "' . $this->escape(json_encode($value)) . '" WHERE `key` = "' . $this->escape($key) . '"'
226:         );
227:     }
228: 
229:     /**
230:      * Helper method for sanitizing input
231:      *
232:      * @param   string
233:      * @return  void
234:      */
235:     private function escape($value) {
236:         return base64_encode($value);
237:     }
238: 
239:     /**
240:      * Helper method for making output human readable
241:      *
242:      * @param   string
243:      * @return  void
244:      */
245:     private function unEscape($value) {
246:         return base64_decode($value);
247:     }
