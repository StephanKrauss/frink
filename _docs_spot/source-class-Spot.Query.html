357:     /**
358:      * Parse array-syntax WHERE conditions and translate them to DBAL QueryBuilder syntax
359:      *
360:      * @param array $where Array of conditions for this clause
361:      * @param bool $useAlias
362:      * @return array SQL fragment strings for WHERE clause
363:      * @throws Exception
364:      */
365:     private function parseWhereToSQLFragments(array $where, $useAlias = true)
366:     {
367:         $builder = $this->builder();
368: 
369:         $sqlFragments = [];
370:         foreach ($where as $column => $value) {
371:             // Column name with comparison operator
372:             $colData = explode(' ', $column);
373:             $operator = isset($colData[1]) ? $colData[1] : '=';
374:             if (count($colData) > 2) {
375:                 $operator = array_pop($colData);
376:                 $colData = [implode(' ', $colData), $operator];
377:             }
378: 
379:             $operator = $this->getWhereOperatorCallable(strtolower($operator));
380:             if (!$operator) {
381:                 throw new Exception("Unsupported operator '" . $operator . "' in WHERE clause");
382:             }
383: 
384:             $col = $colData[0];
385:             if ($useAlias === true) {
386:                 // Prefix column name with alias
387:                 $col = $this->fieldWithAlias($col);
388:             }
389: 
390:             $sqlFragments[] = $operator($builder, $col, $value);
391:         }
392: 
393:         return $sqlFragments;
394:     }
