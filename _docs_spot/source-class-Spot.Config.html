104:     /**
105:      * Taken from PHPUnit 3.4:
106:      * @link http://github.com/sebastianbergmann/phpunit/blob/3.4/PHPUnit/Util/PDO.php
107:      *
108:      * Returns the Data Source Name as a structure containing the various parts of the DSN.
109:      *
110:      * Additional keys can be added by appending a URI query string to the
111:      * end of the DSN.
112:      *
113:      * The format of the supplied DSN is in its fullest form:
114:      * <code>
115:      *  adapter(dbsyntax)://user:password@protocol+host/database?option=8&another=true
116:      * </code>
117:      *
118:      * Most variations are allowed:
119:      * <code>
120:      *  adapter://user:password@protocol+host:110//usr/db_file.db?mode=0644
121:      *  adapter://user:password@host/database_name
122:      *  adapter://user:password@host
123:      *  adapter://user@host
124:      *  adapter://host/database
125:      *  adapter://host
126:      *  adapter(dbsyntax)
127:      *  adapter
128:      * </code>
129:      *
130:      * This function is 'borrowed' from PEAR /DB.php .
131:      *
132:      * @param string $dsn Data Source Name to be parsed
133:      *
134:      * @return array an associative array with the following keys:
135:      *               + adapter:  Database backend used in PHP (mysql, odbc etc.)
136:      *               + dbsyntax: Database used with regards to SQL syntax etc.
137:      *               + protocol: Communication protocol to use (tcp, unix etc.)
138:      *               + host: Host specification (hostname[:port])
139:      *               + dbname: Database to use on the DBMS server
140:      *               + user: User name for login
141:      *               + password: Password for login
142:      */
143:     public static function parseDsn($dsn)
144:     {
145:         if ($dsn == 'sqlite::memory:') {
146:             $dsn = 'sqlite://:memory:';
147:         }
148: 
149:         $parsed = [
150:             'adapter'  => FALSE,
151:             'dbsyntax' => FALSE,
152:             'user'     => FALSE,
153:             'password' => FALSE,
154:             'protocol' => FALSE,
155:             'host'     => FALSE,
156:             'port'     => FALSE,
157:             'socket'   => FALSE,
158:             'dbname'   => FALSE,
159:         ];
160: 
161:         if ( is_array( $dsn ) ) {
162:             $dsn = array_merge( $parsed, $dsn );
163:             if (!$dsn['dbsyntax']) {
164:                 $dsn['dbsyntax'] = $dsn['adapter'];
165:             }
166: 
167:             return $dsn;
168:         }
169: 
170:         // Find phptype and dbsyntax
171:         if ( ( $pos = strpos( $dsn, '://' ) ) !== FALSE ) {
172:             $str = substr( $dsn, 0, $pos );
173:             $dsn = substr( $dsn, $pos + 3 );
174:         } elseif (( $pos = strpos( $dsn, ':' ) ) !== FALSE) {
175:             $str = substr( $dsn, 0, $pos );
176:             $dsn = substr( $dsn, $pos + 1 );
177:         } else {
178:             $str = $dsn;
179:             $dsn = null;
180:         }
181: 
182:         // Get phptype and dbsyntax
183:         // $str => phptype(dbsyntax)
184:         if ( preg_match( '|^(.+?)\((.*?)\)$|', $str, $arr ) ) {
185:             $parsed['adapter']  = $arr[1];
186:             $parsed['dbsyntax'] = !$arr[2] ? $arr[1] : $arr[2];
187:         } else {
188:             $parsed['adapter']  = $str;
189:             $parsed['dbsyntax'] = $str;
190:         }
191: 
192:         if ( !count( $dsn ) ) {
193:             return $parsed;
194:         }
195: 
196:         // Get (if found): user and password
197:         // $dsn => user:password@protocol+host/database
198:         if ( ( $at = strrpos( (string) $dsn, '@' ) ) !== FALSE ) {
199:             $str = substr( $dsn, 0, $at );
200:             $dsn = substr( $dsn, $at + 1 );
201:             if ( ( $pos = strpos( $str, ':' ) ) !== FALSE ) {
202:                 $parsed['user'] = rawurldecode( substr( $str, 0, $pos ) );
203:                 $parsed['password'] = rawurldecode( substr( $str, $pos + 1 ) );
204:             } else {
205:                 $parsed['user'] = rawurldecode( $str );
206:             }
207:         }
208: 
209:         // Find protocol and host
210: 
211:         if ( preg_match( '|^([^(]+)\((.*?)\)/?(.*?)$|', $dsn, $match ) ) {
212:             // $dsn => proto(proto_opts)/database
213:             $proto       = $match[1];
214:             $proto_opts  = $match[2] ? $match[2] : false;
215:             $dsn         = $match[3];
216:         } else {
217:             // $dsn => protocol+host/database (old format)
218:             if ( strpos( $dsn, '+' ) !== FALSE ) {
219:                 list( $proto, $dsn ) = explode( '+', $dsn, 2 );
220:             }
221:             if ( strpos( $dsn, '/' ) !== FALSE ) {
222:                 list( $proto_opts, $dsn ) = explode( '/', $dsn, 2 );
223:             } else {
224:                 $proto_opts = $dsn;
225:                 $dsn = null;
226:             }
227:         }
228: 
229:         // process the different protocol options
230:         $parsed['protocol'] = ( !empty( $proto ) ) ? $proto : 'tcp';
231:         $proto_opts = rawurldecode( $proto_opts );
232:         if ($parsed['protocol'] == 'tcp') {
233:             if ( strpos( $proto_opts, ':' ) !== FALSE ) {
234:                 list( $parsed['host'], $parsed['port'] ) = explode( ':', $proto_opts );
235:             } else {
236:                 $parsed['host'] = $proto_opts;
237:             }
238:         } elseif ($parsed['protocol'] == 'unix') {
239:             $parsed['socket'] = $proto_opts;
240:         }
241: 
242:         // Get dabase if any
243:         // $dsn => database
244:         if ($dsn) {
245:             if ( ( $pos = strpos( $dsn, '?' ) ) === FALSE ) {
246:                 // /database
247:                 $parsed['dbname'] = rawurldecode( $dsn );
248:             } else {
249:                 // /database?param1=value1&param2=value2
250:                 $parsed['dbname'] = rawurldecode( substr( $dsn, 0, $pos ) );
251:                 $dsn = substr( $dsn, $pos + 1 );
252:                 if ( strpos( $dsn, '&') !== FALSE ) {
253:                     $opts = explode( '&', $dsn );
254:                 } else { // database?param1=value1
255:                     $opts = [$dsn];
256:                 }
257:                 foreach ($opts as $opt) {
258:                     list( $key, $value ) = explode( '=', $opt );
259:                     if ( !isset( $parsed[$key] ) ) {
260:                         // don't allow params overwrite
261:                         $parsed[$key] = rawurldecode( $value );
262:                     }
263:                 }
264:             }
265:         }
266: 
267:         // Replace 'adapter' with 'driver' and add 'pdo_'
268:         if (isset($parsed['adapter'])) {
269:             $parsed['driver'] = 'pdo_' . $parsed['adapter'];
270:             unset($parsed['adapter']);
271:         }
272: 
273:         return $parsed;
274:     }
